% Slide 3: Hands-On Exercise
\begin{frame}{Exercise: Bias Detection in Practice}
\begin{columns}[T]
\column{0.48\textwidth}
\textcolor{ForestGreen}{\textbf{Scenario}}

You're reviewing a credit scoring model:

\textcolor{Amber}{Dataset:}
\begin{itemize}
\item 100k loan applications
\item 60\% approved overall
\item Features: income, assets, history
\end{itemize}

\textcolor{Teal}{Performance:}
\begin{itemize}
\item Overall accuracy: 78\%
\item AUC: 0.82
\end{itemize}

\textcolor{DarkTeal}{Demographic Breakdown:}
\begin{itemize}
\item Group A (majority): 65\% approved
\item Group B (minority): 45\% approved
\item Base credit quality similar
\end{itemize}

\column{0.48\textwidth}
\textcolor{ForestGreen}{\textbf{Your Tasks}}

\textcolor{Amber}{1. Identify Issues}
\begin{itemize}
\item What fairness violations?
\item Which metrics to measure?
\item Potential causes?
\end{itemize}

\textcolor{Teal}{2. Diagnose}
\begin{itemize}
\item Data bias?
\item Model bias?
\item Threshold bias?
\end{itemize}

\textcolor{DarkTeal}{3. Propose Solutions}
\begin{itemize}
\item Pre-processing?
\item In-processing?
\item Post-processing?
\item Non-technical interventions?
\end{itemize}

\vspace{0.3cm}
\textcolor{Amber}{\textbf{Discuss:}}\\
Trade-offs between solutions
\end{columns}

\vspace{\fill}
\footnotesize\textcolor{Slate}{Work in groups, present findings}
\end{frame}

% Slide 4: Building an Ethics Review Process
\begin{frame}{Institutional Ethics Review Process}
\begin{columns}[T]
\column{0.48\textwidth}
\textcolor{ForestGreen}{\textbf{Structure}}

\textcolor{Amber}{Ethics Review Board:}
\begin{itemize}
\item Technical experts
\item Domain specialists
\item Ethicists/philosophers
\item Legal counsel
\item Community representatives
\item Rotating membership
\end{itemize}

\vspace{0.3cm}
\textcolor{Teal}{Scope:}
\begin{itemize}
\item All high-risk AI systems
\item Threshold: Impacts on 1000+ people
\item Sensitive domains: Health, justice, employment
\item Customer-facing decisions
\end{itemize}

\column{0.48\textwidth}
\textcolor{ForestGreen}{\textbf{Process}}

\textcolor{DarkTeal}{Stage 1: Initial Review}
\begin{itemize}
\item Submit project proposal
\item Risk assessment
\item 2-week turnaround
\end{itemize}

\textcolor{Amber}{Stage 2: Full Review}
\begin{itemize}
\item Detailed documentation
\item Stakeholder analysis
\item Bias testing results
\item Mitigation plans
\item 6-week evaluation
\end{itemize}

\textcolor{Teal}{Stage 3: Monitoring}
\begin{itemize}
\item Quarterly reports
\item Incident tracking
\item Annual re-certification
\end{itemize}

\vspace{0.3cm}
\textcolor{DarkTeal}{\textbf{Authority:}}\\
Board can block deployment
\end{columns}

\vspace{\fill}
\footnotesize\textcolor{Slate}{Model after institutional review boards (IRBs) in research}
\end{frame}

% Slide 6: EU AI Act Deep Dive
\begin{frame}{EU AI Act: What You Need to Know}
\begin{columns}[T]
\column{0.48\textwidth}
\textcolor{ForestGreen}{\textbf{Risk Categories}}

\textcolor{Amber}{Unacceptable Risk (Banned):}
\begin{itemize}
\item Social scoring
\item Manipulative AI
\item Real-time biometric surveillance (limited exceptions)
\item Emotion recognition in workplaces/schools
\end{itemize}

\textcolor{Teal}{High Risk (Regulated):}
\begin{itemize}
\item Critical infrastructure
\item Education/employment
\item Law enforcement
\item Border control
\item Administration of justice
\end{itemize}

\textcolor{DarkTeal}{Limited/Minimal Risk:}
\begin{itemize}
\item Transparency requirements
\item Self-regulation
\end{itemize}

\column{0.48\textwidth}
\textcolor{ForestGreen}{\textbf{High-Risk Requirements}}

\textcolor{Amber}{Before Deployment:}
\begin{itemize}
\item Risk management system
\item Data governance
\item Technical documentation
\item Record-keeping
\item Transparency
\item Human oversight
\item Accuracy requirements
\item Cybersecurity
\item Conformity assessment
\end{itemize}

\textcolor{Teal}{After Deployment:}
\begin{itemize}
\item Post-market monitoring
\item Incident reporting
\item Corrective actions
\end{itemize}

Fully applicable: August 2026
