% Slide 5: Ethical Frameworks
\begin{frame}{Ethical Frameworks for AI}
\begin{columns}[T]
\column{0.48\textwidth}
\textcolor{ForestGreen}{\textbf{Consequentialist}}

Focus on outcomes:
\begin{itemize}
\item Maximize benefit
\item Minimize harm
\item Utilitarian calculus
\item Risk-benefit analysis
\end{itemize}

\vspace{0.3cm}
\textcolor{Amber}{\textbf{Example:}}\\
Deploy system if overall benefit outweighs harm to affected groups

\vspace{0.3cm}
\textcolor{Teal}{\textbf{Deontological}}

Focus on duties:
\begin{itemize}
\item Respect autonomy
\item Honor rights
\item Follow rules
\item Categorical imperatives
\end{itemize}

\column{0.48\textwidth}
\textcolor{DarkTeal}{\textbf{Virtue Ethics}}

Focus on character:
\begin{itemize}
\item Cultivate wisdom
\item Practice fairness
\item Show compassion
\item Demonstrate integrity
\end{itemize}

\vspace{0.3cm}
\textcolor{Amber}{\textbf{Example:}}\\
Ask ``What would a fair person do?''

\vspace{0.3cm}
\textcolor{ForestGreen}{\textbf{Care Ethics}}

Focus on relationships:
\begin{itemize}
\item Understand context
\item Value relationships
\item Address vulnerability
\item Emphasize empathy
\end{itemize}
\end{columns}

\vspace{\fill}
\footnotesize\textcolor{Slate}{No single framework is perfect -- combine perspectives for robust ethics}
\end{frame}
