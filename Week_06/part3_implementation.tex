% Local vs Cloud
\begin{frame}{Local vs Cloud Deployment}
\begin{columns}[T]
\column{0.48\textwidth}
\textcolor{mlblue}{\textbf{Cloud APIs}}

\small
\textcolor{mlgreen}{Pros:}
\begin{itemize}
\item No infrastructure needed
\item Latest models instantly
\item Infinite scale
\item Pay-per-use
\end{itemize}

\textcolor{mlred}{Cons:}
\begin{itemize}
\item Ongoing costs
\item Internet dependency
\item Data privacy concerns
\item Vendor lock-in
\end{itemize}

\textcolor{mlorange}{Best for:} Prototypes, variable load

\column{0.48\textwidth}
\textcolor{mlpurple}{\textbf{Local Models}}

\small
\textcolor{mlgreen}{Pros:}
\begin{itemize}
\item Full control
\item No usage costs
\item Data stays private
\item Customization possible
\end{itemize}

\textcolor{mlred}{Cons:}
\begin{itemize}
\item Hardware requirements
\item Setup complexity
\item Model limitations
\item Maintenance burden
\end{itemize}

\textcolor{mlorange}{Best for:} Production, sensitive data
\end{columns}

\vspace{\fill}
\footnotesize\textcolor{mlgray}{Hybrid approach often optimal: prototype cloud, deploy local}
\end{frame}

% Fine-tuning Strategies
\begin{frame}{Fine-tuning for Your Domain}
\begin{columns}[T]
\column{0.48\textwidth}
\textcolor{mlblue}{\textbf{When to Fine-tune}}

\small
\begin{itemize}
\item Domain-specific language
\item Consistent style needed
\item Proprietary knowledge
\item Performance optimization
\item Cost reduction at scale
\end{itemize}

\vspace{0.3cm}
\textcolor{mlorange}{\textbf{Techniques:}}

\small
\begin{itemize}
\item \textbf{Full fine-tuning:} All parameters
\item \textbf{LoRA:} Low-rank adaptation
\item \textbf{QLoRA:} Quantized LoRA
\item \textbf{Prefix tuning:} Soft prompts
\item \textbf{Adapter layers:} Modular
\end{itemize}

\column{0.48\textwidth}
\textcolor{mlgreen}{\textbf{Process}}

\small
\begin{enumerate}
\item Collect domain data (1000+ examples)
\item Clean and format
\item Choose base model
\item Set hyperparameters
\item Train (4-24 hours)
\item Evaluate thoroughly
\item Deploy with monitoring
\end{enumerate}

\vspace{0.3cm}
\textcolor{mlpurple}{\textbf{Costs:}}
\begin{itemize}
\item GPT-3.5: \$0.008/1K tokens
\item Llama 2: Free (compute only)
\item Custom: \$500-5000 typical
\end{itemize}
\end{columns}

\vspace{\fill}
\footnotesize\textcolor{mlgray}{Fine-tuning can 10x performance for specific tasks}
\end{frame}
