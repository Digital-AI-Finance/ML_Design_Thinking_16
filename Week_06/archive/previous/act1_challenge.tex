% Slide 2: Build concept of Traditional Prototyping Pipeline
\begin{frame}[t]{The Traditional Prototyping Pipeline}
\begin{columns}[T]
\column{0.55\textwidth}
\textcolor{mlblue}{\textbf{The Sequential Workflow}}

\small
\textbf{Stage 1: Sketching} (4-6 hours)
\begin{itemize}
\item Hand-drawn wireframes
\item User flow diagrams
\item Feature brainstorming
\end{itemize}

\textbf{Stage 2: Design} (3-5 days)
\begin{itemize}
\item Visual identity creation
\item High-fidelity mockups
\item Asset generation
\item \textcolor{mlred}{Requires: Designer (\$2K-5K)}
\end{itemize}

\textbf{Stage 3: Copywriting} (2-3 days)
\begin{itemize}
\item Landing page text
\item Feature descriptions
\item Call-to-action
\item \textcolor{mlred}{Requires: Writer (\$500-1K)}
\end{itemize}

\textbf{Stage 4: Development} (1-2 weeks)
\begin{itemize}
\item Code structure
\item Functionality
\item Integration
\item \textcolor{mlred}{Requires: Developer (\$5K-10K)}
\end{itemize}

\column{0.43\textwidth}
\textcolor{mlorange}{\textbf{The Hard Math}}

\small
\textbf{Total Timeline:}\\
2-4 weeks minimum

\vspace{0.3cm}
\textbf{Total Cost:}\\
\$10,000 - \$50,000

\vspace{0.3cm}
\textbf{Team Size:}\\
3-5 specialists

\vspace{0.3cm}
\textbf{Dependencies:}\\
Sequential (can't parallelize)

\vspace{0.5cm}
\begin{tcolorbox}[colback=mlyellow!20, colframe=mlorange]
\centering
\small
\textbf{Reality:} Each iteration takes another\\2-4 weeks and \$10K-50K
\end{tcolorbox}

\vspace{0.5cm}
\textcolor{mlred}{\textbf{The Killer Problem:}}\\
\small
By the time you have prototype,\\
market has moved on
\end{columns}

\bottomnote{Traditional prototyping = high cost, long timeline, limited iterations}
\end{frame}

% Slide 3: Build "Prototype" concept from scratch
\begin{frame}[t]{What IS a Prototype? (Building from Scratch)}
\begin{columns}[T]
\column{0.48\textwidth}
\textcolor{mlblue}{\textbf{Human Analogy: Cooking}}

\small
You want to cook new dish:

\vspace{0.3cm}
\textbf{Don't start with:}\\
\textcolor{mlred}{X} Full restaurant service\\
\textcolor{mlred}{X} Perfect plating\\
\textcolor{mlred}{X} 5-course meal

\vspace{0.3cm}
\textbf{Start with:}\\
\textcolor{mlgreen}{\checkmark} Taste test with friends\\
\textcolor{mlgreen}{\checkmark} Basic version\\
\textcolor{mlgreen}{\checkmark} Learn what works

\vspace{0.5cm}
\textcolor{mlpurple}{\textbf{Computer Equivalent:}}

\small
Testable version of idea that:
\begin{itemize}
\item Demonstrates core concept
\item Gets real feedback
\item Costs little to make
\item Easy to change
\end{itemize}

\column{0.48\textwidth}
\textcolor{mlorange}{\textbf{The Fidelity Spectrum}}

\small
\textbf{Paper Sketch} (1 hour)\\
Fidelity: 10\%\\
Learning: Basic flow\\
Cost: \$0

\vspace{0.3cm}
\textbf{Clickable Mockup} (1 day)\\
Fidelity: 40\%\\
Learning: User interaction\\
Cost: \$50-500

\vspace{0.3cm}
\textbf{Working MVP} (1-2 weeks)\\
Fidelity: 70\%\\
Learning: Real usage\\
Cost: \$5K-20K

\vspace{0.3cm}
\textbf{Production App} (3-6 months)\\
Fidelity: 100\%\\
Learning: Market fit\\
Cost: \$100K+

\vspace{0.5cm}
\begin{tcolorbox}[colback=mlgreen!10, colframe=mlgreen]
\centering
\small
\textbf{Key Insight:} Fidelity vs Speed tradeoff
\end{tcolorbox}

\bottomnote{A prototype is a testable version optimized for learning, not perfection}
\end{frame}

% Slide 4: Build "Creation Bottleneck" concept
\begin{frame}[t]{The Creation Bottleneck (Why Building Is Hard)}
\begin{columns}[T]
\column{0.48\textwidth}
\textcolor{mlpurple}{\textbf{Three Bottlenecks}}

\small
\textbf{1. The Skill Gap}
\begin{itemize}
\item Design requires aesthetics
\item Code requires programming
\item Copy requires writing
\item Video requires editing
\end{itemize}
\textcolor{mlred}{Problem:} Need 4+ specialized skills

\vspace{0.3cm}
\textbf{2. The Time Constraint}
\begin{itemize}
\item Design: 20-40 hours
\item Development: 40-80 hours
\item Content: 10-20 hours
\item Testing: 10-20 hours
\end{itemize}
\textcolor{mlred}{Problem:} 80-160 hours total

\vspace{0.3cm}
\textbf{3. The Iteration Cost}
\begin{itemize}
\item Each change = hours of work
\item Must coordinate team
\item Testing reveals more issues
\item Changes compound
\end{itemize}
\textcolor{mlred}{Problem:} Change is expensive

\column{0.48\textwidth}
\textcolor{mlblue}{\textbf{The Exponential Problem}}

\small
Cost = Skills $\times$ Time $\times$ Iterations

\vspace{0.5cm}
\textbf{Example Calculation:}

\small
For EcoTrack app prototype:
\begin{itemize}
\item Skills needed: 4 (design, code, copy, video)
\item Time per skill: 20 hours average
\item Iterations: 3 typical
\end{itemize}

\vspace{0.3cm}
Total effort: $4 \times 20 \times 3 = 240$ hours

\vspace{0.3cm}
At \$100/hour: \textcolor{mlred}{\textbf{\$24,000}}

\vspace{0.5cm}
\begin{tcolorbox}[colback=mlorange!20, colframe=mlorange]
\centering
\textbf{The Real Cost:}\\
Most ideas die before prototyping\\
because the bottleneck is too expensive
\end{tcolorbox}

\vspace{0.3cm}
\textcolor{mlgray}{\small
\textit{``The best prototype is the one you actually build''}\\
- Every designer who's seen ideas die in planning}
\end{columns}

\bottomnote{Creation bottleneck = Skills $\times$ Time $\times$ Iterations makes most ideas unviable}
\end{frame}

% Slide 5: Quantify the Innovation Loss (The Scale of the Problem)
\begin{frame}[t]{Quantifying the Innovation Loss}
\begin{columns}[T]
\column{0.55\textwidth}
\textcolor{mlred}{\textbf{The Harsh Math of Innovation}}

\small
Average entrepreneur has:
\begin{itemize}
\item \textbf{100 ideas} per year
\item \textbf{Time for 3 prototypes} (bottleneck limit)
\item \textbf{Success rate:} 33\% of tested ideas work
\end{itemize}

\vspace{0.5cm}
\Large
\textcolor{mlpurple}{\textbf{The Information Loss:}}

\normalsize
\vspace{0.3cm}
\begin{center}
\begin{tabular}{lcc}
\toprule
\textbf{Stage} & \textbf{Count} & \textbf{Loss} \\
\midrule
Ideas generated & 100 & - \\
\textcolor{mlred}{Ideas prototyped} & \textcolor{mlred}{3} & \textcolor{mlred}{-97\%} \\
Ideas that succeed & 1 & -66\% \\
\bottomrule
\end{tabular}
\end{center}

\vspace{0.5cm}
\textcolor{mlorange}{\textbf{What This Means:}}\\
\small
If all 100 were tested: $100 \times 0.33 = 33$ successes\\
Actually tested: $3 \times 0.33 = 1$ success\\
\textbf{Lost potential:} \textcolor{mlred}{32 successful products never built}

\column{0.43\textwidth}
\textcolor{mlblue}{\textbf{Information Theory View}}

\small
\textbf{Prototyping Bandwidth:}\\
3 prototypes/year = 3 bits of information

\vspace{0.3cm}
\textbf{Idea Generation Bandwidth:}\\
100 ideas/year = 100 bits of information

\vspace{0.3cm}
\textbf{Bandwidth Mismatch:}\\
$100 - 3 = 97$ bits lost to bottleneck

\vspace{0.5cm}
\begin{tcolorbox}[colback=mlred!20, colframe=mlred]
\centering
\textbf{97\% of innovation}\\
\textbf{dies at the prototyping stage}
\end{tcolorbox}

\vspace{0.5cm}
\textcolor{mlgreen}{\textbf{The Dream:}}\\
\small
What if you could test all 100 ideas?\\
What if prototyping was free?\\
What if iteration was instant?

\vspace{0.5cm}
\begin{tcolorbox}[colback=mlblue!10, colframe=mlblue]
\centering
\small
\textbf{Forward Question:}\\
Can we eliminate the bottleneck?
\end{tcolorbox}
\end{columns}

\bottomnote{The prototyping bottleneck kills 97\% of ideas before they're tested - this is the challenge}
\end{frame}
