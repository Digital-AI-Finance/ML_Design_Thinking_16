% Slide 11: Diagnosis - Root Cause Analysis
\begin{frame}[t]{Diagnosing the Root Cause}
\begin{columns}[T]
\column{0.48\textwidth}
\textcolor{mlblue}{\textbf{Trace Specific Example}}

\small
\textbf{Task:} Create EcoTrack homepage

\vspace{0.3cm}
\textbf{Prompt given:}\\
\textit{``Create a homepage for EcoTrack''}

\vspace{0.5cm}
\textcolor{mlgreen}{\textbf{What AI Captured:}}
\begin{itemize}
\item Generic app homepage structure
\item Standard navigation menu
\item Hero section with image
\item Feature list layout
\item Call-to-action button
\end{itemize}

\vspace{0.3cm}
\textcolor{mlgray}{Result: Looks like every other app homepage}

\column{0.48\textwidth}
\textcolor{mlred}{\textbf{What AI Missed:}}

\small
\begin{itemize}
\item \textbf{Brand Identity}\\
EcoTrack's earth-tone palette\\
Minimalist, trustworthy style

\item \textbf{Target Audience}\\
Environmentally-conscious millennials\\
Professional tone, not playful

\item \textbf{Technical Constraints}\\
Must be mobile-first\\
Load time < 2 seconds

\item \textbf{Domain Knowledge}\\
Carbon footprint terminology\\
Scientific accuracy required

\item \textbf{Integration}\\
Must connect to carbon API\\
Match existing logo colors
\end{itemize}

\vspace{0.3cm}
\textcolor{mlred}{\textbf{Root Cause:}}\\
AI has \textcolor{mlred}{\textbf{no context}} about:\\
our specific project
\end{columns}

\vspace{0.3cm}
\begin{center}
\begin{tcolorbox}[colback=mlorange!20, colframe=mlorange, width=0.85\textwidth]
\centering
\textbf{The Mismatch:} 90\% of domain-specific requirements lost\\
AI generates from general patterns, not project-specific context
\end{tcolorbox}
\end{center}

\bottomnote{Diagnosis: AI lacks project context, constraints, brand guidelines, and domain knowledge}
\end{frame}
