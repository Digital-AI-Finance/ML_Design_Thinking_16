% Step-by-Step Guide
\begin{frame}[fragile]{Step-by-Step: Prompt Engineering}
\begin{columns}[T]
\column{0.48\textwidth}
\textcolor{mlblue}{\textbf{Step 1: Ideation Prompt}}

\footnotesize
\texttt{Generate 10 innovative product ideas}\\
\texttt{that solve [problem] for [audience].}\\
\texttt{For each idea provide:}\\
\texttt{- Name}\\
\texttt{- One-line description}\\
\texttt{- Key differentiator}\\
\texttt{- Target market size}

\vspace{0.2cm}
\textcolor{mlorange}{\textbf{Step 2: Refinement}}

\footnotesize
\texttt{Take idea \#3 and expand:}\\
\texttt{- Problem it solves (specific)}\\
\texttt{- Solution approach}\\
\texttt{- 5 key features}\\
\texttt{- Revenue model}\\
\texttt{- Competitive advantage}

\column{0.48\textwidth}
\textcolor{mlgreen}{\textbf{Step 3: Visual Generation}}

\footnotesize
\texttt{Create a product mockup:}\\
\texttt{"Minimalist app interface for}\\
\texttt{[product], clean design,}\\
\texttt{modern UI, mobile screen,}\\
\texttt{professional, Figma style"}

\vspace{0.2cm}
\textcolor{mlpurple}{\textbf{Step 4: Copy Creation}}

\footnotesize
\texttt{Write landing page copy:}\\
\texttt{- Hero headline (7 words)}\\
\texttt{- Subheadline (15 words)}\\
\texttt{- 3 benefit statements}\\
\texttt{- Call to action}\\
\texttt{Target emotion: excitement}
\end{columns}

\vspace{\fill}
\footnotesize\textcolor{mlgray}{Good prompts = Good outputs}
\end{frame}

% Common Pitfalls
\begin{frame}{Avoiding Common Pitfalls}
\begin{columns}[T]
\column{0.48\textwidth}
\textcolor{mlred}{\textbf{Mistakes to Avoid}}

\small
\begin{itemize}
\item Vague prompts $\rightarrow$ Vague outputs
\item No iteration $\rightarrow$ Mediocre results
\item Ignoring constraints $\rightarrow$ Unusable
\item Over-relying on AI $\rightarrow$ Generic
\item No human review $\rightarrow$ Errors
\end{itemize}

\vspace{0.3cm}
\textcolor{mlorange}{\textbf{Quality Checklist:}}
\begin{itemize}
\item[$\square$] Is it original?
\item[$\square$] Does it solve a real problem?
\item[$\square$] Is it feasible?
\item[$\square$] Would you use it?
\end{itemize}

\column{0.48\textwidth}
\textcolor{mlgreen}{\textbf{Pro Tips}}

\small
\begin{itemize}
\item Start with 10 variations
\item Combine best elements
\item Add specific constraints
\item Use reference examples
\item Iterate at least 3 times
\end{itemize}

\vspace{0.3cm}
\textcolor{mlpurple}{\textbf{Time Allocation:}}
\begin{itemize}
\item Ideation: 5 min
\item Generation: 10 min
\item Refinement: 10 min
\item Polish: 5 min
\end{itemize}
\end{columns}

\vspace{\fill}
\footnotesize\textcolor{mlgray}{The difference between good and great is iteration}
\end{frame}

% Ethics and Attribution
\begin{frame}{Responsible AI Use}
\begin{columns}[T]
\column{0.48\textwidth}
\textcolor{mlblue}{\textbf{Ethical Guidelines}}

\small
\begin{itemize}
\item Always disclose AI use
\item Verify all facts
\item Check for bias
\item Respect IP rights
\item Maintain human oversight
\end{itemize}

\vspace{0.3cm}
\textcolor{mlred}{\textbf{Legal Considerations:}}
\begin{itemize}
\item Copyright unclear
\item Terms of service vary
\item Attribution required
\item Commercial use restrictions
\end{itemize}

\column{0.48\textwidth}
\textcolor{mlgreen}{\textbf{Best Practices}}

\small
\begin{itemize}
\item Credit: "Created with AI assistance"
\item Document your process
\item Keep human in the loop
\item Test with real users
\item Monitor for harmful content
\end{itemize}

\vspace{0.3cm}
\textcolor{mlpurple}{\textbf{Remember:}}\\
\small AI is a tool, not a replacement for human judgment
\end{columns}

\vspace{\fill}
