\begin{frame}{Step-by-Step Implementation}
\begin{columns}[T]
\column{0.48\textwidth}
\textcolor{mlblue}{\textbf{Step 1: Data Preparation}}
\footnotesize
\texttt{
\# Clean text\\
def preprocess(text):\\
~~~~\# Lowercase\\
~~~~\# Remove special chars\\
~~~~\# Tokenize\\
~~~~\# Remove stopwords\\
~~~~\# Lemmatize\\
~~~~return tokens\\
\\
docs = df['description'].apply(preprocess)
}

\vspace{0.2cm}
\textcolor{mlorange}{\textbf{Step 2: Build Model}}
\footnotesize
\texttt{
\# Create dictionary\\
dictionary = corpora.Dictionary(docs)\\
corpus = [dictionary.doc2bow(doc)\\
~~~~~~~~~~for doc in docs]\\
\\
\# Train LDA\\
lda = models.LdaModel(\\
~~~~corpus, num\_topics=15,\\
~~~~id2word=dictionary\\
)
}

\column{0.48\textwidth}
\textcolor{mlgreen}{\textbf{Step 3: Extract Insights}}
\footnotesize
\texttt{
\# Get topics\\
for idx, topic in lda.print\_topics():\\
~~~~print(f'Topic \{idx\}: \{topic\}')\\
\\
\# Document-topic distribution\\
doc\_topics = []\\
for doc in corpus:\\
~~~~doc\_topics.append(\\
~~~~~~~~lda.get\_document\_topics(doc)\\
~~~~)
}

\vspace{0.2cm}
\textcolor{mlpurple}{\textbf{Step 4: Visualize}}
\footnotesize
\texttt{
\# Create visualization\\
import pyLDAvis.gensim\\
vis = pyLDAvis.gensim.prepare(\\
~~~~lda, corpus, dictionary\\
)\\
pyLDAvis.display(vis)

% Common Challenges
\begin{frame}{Common Challenges \& Solutions}
\begin{columns}[T]
\column{0.48\textwidth}
\textcolor{mlred}{\textbf{Challenge 1: Messy Topics}}

\small
\textbf{Problem:} Topics don't make sense

\textbf{Solution:}
\begin{itemize}
\item Better preprocessing
\item Adjust number of topics
\item Remove more stopwords
\item Try different algorithms
\end{itemize}

\vspace{0.3cm}
\textcolor{mlred}{\textbf{Challenge 2: Overlapping Topics}}

\small
\textbf{Problem:} Topics too similar

\textbf{Solution:}
\begin{itemize}
\item Reduce number of topics
\item Increase alpha parameter
\item Use hierarchical models
\end{itemize}

\column{0.48\textwidth}
\textcolor{mlred}{\textbf{Challenge 3: Unactionable Insights}}

\small
\textbf{Problem:} Topics not useful

\textbf{Solution:}
\begin{itemize}
\item Add domain knowledge
\item Filter by relevance
\item Combine with other data
\item Involve stakeholders
\end{itemize}

\vspace{0.3cm}
\textcolor{mlred}{\textbf{Challenge 4: Scale Issues}}

\small
\textbf{Problem:} Too slow on big data

\textbf{Solution:}
\begin{itemize}
\item Sample documents
\item Use online learning
\item Parallelize processing
\item Cloud computing
\end{itemize}
\end{columns}
\end{frame}
