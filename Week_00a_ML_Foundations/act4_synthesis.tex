% Slide 1: The Complete ML Pipeline
\begin{frame}{The Complete Machine Learning Pipeline}
\begin{center}
\includegraphics[width=0.95\textwidth]{charts/ml_pipeline_complete.pdf}
\end{center}

\vspace{0.5em}
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{Data Phase:}
\begin{enumerate}
\item Collect raw data (web scraping, sensors, databases)
\item Clean data (handle missing values, outliers)
\item Split into train/validation/test (70\%/15\%/15\%)
\end{enumerate}

\textbf{Model Phase:}
\begin{enumerate}
\setcounter{enumi}{3}
\item Choose algorithm (linear, SVM, neural network)
\item Train model (optimize parameters)
\item Validate (tune hyperparameters, prevent overfitting)
\end{enumerate}
\end{column}

\begin{column}{0.48\textwidth}
\textbf{Deployment Phase:}
\begin{enumerate}
\setcounter{enumi}{6}
\item Test on held-out data (measure real performance)
\item Deploy to production (API, edge device, cloud)
\item Monitor performance (detect data drift)
\item Retrain periodically (maintain accuracy)
\end{enumerate}

\vspace{0.3em}
\keypoint{Machine learning is a continuous cycle, not a one-time process}
\end{column}
\end{columns}

\vspace{0.5em}
\bottomnote{Every step is critical - failure at any stage leads to poor real-world performance}
\end{frame}

% Slide 2: Decision Tree - When to Use Each Approach
\begin{frame}{Choosing the Right Approach: A Decision Guide}
\twocolslide{
\centering
\includegraphics[width=0.95\textwidth]{charts/algorithm_decision_tree.pdf}

\vspace{0.5em}
\textbf{Start here:}

\begin{itemize}
\item \textbf{Data size:} < 1K, 1K-100K, or > 100K samples?
\item \textbf{Interpretability:} Do you need to explain decisions?
\item \textbf{Data type:} Tabular, images, text, or time series?
\item \textbf{Linearity:} Are relationships linear or nonlinear?
\end{itemize}
}{
\Large\textbf{Quick Reference Guide}
\normalsize
\vspace{0.5em}

\textbf{Use Linear/Polynomial Features when:}
\begin{itemize}
\item Small dataset (< 1000 samples)
\item Need interpretability (coefficients matter)
\item Relationship is approximately linear
\item Example: House prices, simple predictions
\end{itemize}

\textbf{Use SVM (RBF kernel) when:}
\begin{itemize}
\item Medium dataset (1K-100K samples)
\item Nonlinear but structured patterns
\item Want guaranteed global optimum
\item Example: Text classification, medical diagnosis
\end{itemize}

\textbf{Use Neural Networks when:}
\begin{itemize}
\item Large dataset (> 100K samples)
\item Complex patterns (images, audio, text)
\item Accuracy is top priority
\item Have computational resources
\item Example: Computer vision, speech recognition
\end{itemize}

\bottomnote{Often best to try multiple approaches and compare}
}
\end{frame}
