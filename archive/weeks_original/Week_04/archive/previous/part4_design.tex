\begin{frame}{Confidence Scores and Uncertainty}
\Large\textbf{Communicating Model Certainty}
\normalsize

\vspace{0.5em}

\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{Probability vs Confidence:}
\begin{center}
\begin{tikzpicture}[scale=0.7]
% High confidence
\draw[thick,->] (0,3) -- (5,3);
\draw[fill=mlgreen!60] (4,2.8) rectangle (4.8,3.2);
\node[left] at (0,3) {High Conf:};
\node[right] at (5,3) {85\% $\pm$ 3\%};

% Medium confidence
\draw[thick,->] (0,2) -- (5,2);
\draw[fill=mlyellow!60] (2.5,1.8) rectangle (3.5,2.2);
\node[left] at (0,2) {Med Conf:};
\node[right] at (5,2) {60\% $\pm$ 15\%};

% Low confidence
\draw[thick,->] (0,1) -- (5,1);
\draw[fill=mlred!60] (1,0.8) rectangle (4,1.2);
\node[left] at (0,1) {Low Conf:};
\node[right] at (5,1) {55\% $\pm$ 30\%};
\end{tikzpicture}
\end{center}

\vspace{0.5em}
\textbf{Sources of Uncertainty:}
\begin{itemize}
\item Limited training data
\item Feature noise
\item Model limitations
\item Distribution shift
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\textbf{Design Guidelines:}
\begin{itemize}
\item Show confidence bands
\item Use color coding
\item Provide explanations
\item Suggest actions based on certainty
\end{itemize}

\vspace{0.5em}
\textbf{User Actions by Confidence:}
\begin{itemize}
\item \textcolor{mlgreen}{High}: Proceed with plan
\item \textcolor{mlyellow}{Medium}: Gather more data
\item \textcolor{mlred}{Low}: Seek expert input
\end{itemize}

\vspace{0.5em}
\begin{tcolorbox}[colback=mlblue!10,colframe=mlblue]
Users trust systems that acknowledge uncertainty
\end{tcolorbox}
\end{column}
\end{columns}
\end{frame}
