% What is Topic Modeling?
\begin{frame}{What is Topic Modeling?}
\begin{columns}[T]
\column{0.48\textwidth}
\textcolor{mlblue}{\textbf{Core Concept}}

\normalsize
Finding hidden themes in large text collections

\vspace{0.5cm}
\small
\textcolor{mlorange}{\textbf{Key Assumptions:}}
\begin{enumerate}
\item Documents contain multiple topics
\item Topics are probability distributions over words
\item Words can belong to multiple topics
\end{enumerate}

\column{0.48\textwidth}
\includegraphics[width=\textwidth]{topic_word_distribution.pdf}
\end{columns}

\vspace{\fill}
\footnotesize\textcolor{mlgray}{Each document is a mixture of topics, each topic is a mixture of words}
\end{frame}

% Types of Topic Models
\begin{frame}{Topic Modeling Approaches}
\begin{columns}[T]
\column{0.48\textwidth}
\textcolor{mlblue}{\textbf{Probabilistic Models}}

\small
\textbf{LDA (Latent Dirichlet Allocation)}
\begin{itemize}
\item Bayesian approach
\item Interpretable topics
\item Industry standard
\end{itemize}

\textbf{LSA (Latent Semantic Analysis)}
\begin{itemize}
\item Matrix factorization
\item Fast computation
\item Good for similarity
\end{itemize}

\column{0.48\textwidth}
\textcolor{mlorange}{\textbf{Modern Approaches}}

\small
\textbf{NMF (Non-negative Matrix Factorization)}
\begin{itemize}
\item Parts-based representation
\item Sparse, interpretable
\item Good for short texts
\end{itemize}

\textbf{BERTopic}
\begin{itemize}
\item Transformer-based
\item Contextual understanding
\item State-of-the-art accuracy
\end{itemize}
\end{columns}

\vspace{\fill}
\footnotesize\textcolor{mlgray}{Choose based on: data size, interpretability needs, computational resources}
\end{frame}
