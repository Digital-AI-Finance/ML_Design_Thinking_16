% Key Concepts to Remember
\begin{frame}{Key Concepts: Building Blocks}
\begin{columns}[T]
\column{0.48\textwidth}
\textcolor{mlblue}{\textbf{Prompting}}

\small
The art of instructing AI models
\begin{itemize}
\item Context setting
\item Role definition
\item Output formatting
\item Iterative refinement
\end{itemize}

\vspace{0.3cm}
\textcolor{mlorange}{\textbf{Temperature \& Sampling}}

\small
Controlling creativity vs consistency
\begin{itemize}
\item Temperature: 0.0 - 2.0
\item Top-p: Nucleus sampling
\item Top-k: Vocabulary limiting
\end{itemize}

\column{0.48\textwidth}
\textcolor{mlgreen}{\textbf{Context Windows}}

\small
Input/output limitations
\begin{itemize}
\item GPT-4: 128K tokens
\item Claude: 200K tokens
\item Gemini: 1M tokens
\item Local models: 4-32K
\end{itemize}

\vspace{0.3cm}
\textcolor{mlpurple}{\textbf{Fine-tuning vs RAG}}

\small
Customization approaches
\begin{itemize}
\item Fine-tuning: Model adaptation
\item RAG: Retrieval augmented
\item LoRA: Efficient tuning
\item Prompt tuning: Soft prompts
\end{itemize}
\end{columns}

\vspace{\fill}
\footnotesize\textcolor{mlgray}{Master these concepts to unlock GenAI potential}
\end{frame}
