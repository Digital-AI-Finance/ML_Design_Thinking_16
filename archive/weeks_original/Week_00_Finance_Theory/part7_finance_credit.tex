\begin{frame}{Credit Scoring: From FICO to Machine Learning}
\begin{columns}[T]
\begin{column}{0.55\textwidth}
\Large\textbf{Probability of Default Modeling}
\normalsize
\vspace{0.5em}

\textbf{Traditional FICO Score:}
\begin{itemize}
\item Payment history: 35\%
\item Amounts owed: 30\%
\item Length of history: 15\%
\item Credit mix: 10\%
\item New credit: 10\%
\end{itemize}

Linear scorecard: 300-850 range

\textbf{Logistic Regression Baseline:}
$$PD = \frac{1}{1 + e^{-(\beta_0 + \beta^TX)}}$$

where $X$ includes debt-to-income, payment history, utilization

\textbf{ML Enhancements:}
\begin{itemize}
\item \textbf{Random Forest:} Non-linear interactions
\item \textbf{XGBoost:} Best predictive performance
\item \textbf{Neural Networks:} Deep feature learning
\item \textbf{Ensemble:} Combine multiple models
\end{itemize}
\end{column}

\begin{frame}{Loss Given Default (LGD) and Recovery Modeling}
\begin{columns}[T]
\begin{column}{0.55\textwidth}
\Large\textbf{Expected Loss Framework}
\normalsize
\vspace{0.5em}

\textbf{Expected Loss Decomposition:}
$$\text{EL} = \text{EAD} \times \text{PD} \times \text{LGD}$$

\begin{itemize}
\item \textbf{EAD:} Exposure at Default
\item \textbf{PD:} Probability of Default
\item \textbf{LGD:} Loss Given Default (1 - Recovery Rate)
\end{itemize}

\textbf{LGD Modeling Challenges:}
\begin{itemize}
\item Bimodal distribution (full loss or high recovery)
\item Limited default data
\item Collateral valuation uncertainty
\item Economic cycle dependencies
\end{itemize}

\textbf{ML Approaches:}
\begin{enumerate}
\item \textbf{Two-stage:} Logistic (default?) + Regression (LGD)
\item \textbf{Beta regression:} LGD $\in$ (0,1) naturally
\item \textbf{Quantile regression:} Estimate LGD distribution
\item \textbf{Survival analysis:} Time-to-recovery modeling
\end{enumerate}
\end{column}

\begin{frame}{Fraud Detection with ML}
\begin{columns}[T]
\begin{column}{0.55\textwidth}
\Large\textbf{Real-Time Transaction Monitoring}
\normalsize
\vspace{0.5em}

\textbf{Fraud Detection Pipeline:}
\begin{enumerate}
\item Feature extraction from transaction
\item Real-time scoring (< 100ms)
\item Rule engine + ML model ensemble
\item Human review for high-risk cases
\end{enumerate}

\textbf{Key Features:}
\begin{itemize}
\item \textbf{Transaction:} Amount, location, merchant, time
\item \textbf{Behavioral:} Deviation from normal patterns
\item \textbf{Historical:} Past fraud, dispute history
\item \textbf{Network:} Graph-based (connected accounts)
\end{itemize}

\textbf{Class Imbalance Problem:}
\begin{itemize}
\item Fraud rate: 0.1-0.5\% of transactions
\item SMOTE oversampling for minority class
\item Cost-sensitive learning (FP vs FN)
\item Precision-recall optimization
\end{itemize}
\end{column}

\begin{column}{0.43\textwidth}
\textbf{ML Techniques:}

\textcolor{finblue}{\textbf{Supervised Models:}}
\begin{itemize}
