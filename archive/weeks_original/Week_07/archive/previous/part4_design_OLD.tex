% Slide 3: Transparency & Consent
\begin{frame}{User Transparency and Consent}
\begin{columns}[T]
\column{0.48\textwidth}
\textcolor{ForestGreen}{\textbf{Transparency Layers}}

\textcolor{Amber}{Level 1: Existence}
\begin{itemize}
\item Disclose AI is being used
\item Not a human interaction
\item System capabilities
\end{itemize}

\textcolor{Teal}{Level 2: Process}
\begin{itemize}
\item How decisions are made
\item What data is used
\item Who has access
\end{itemize}

\textcolor{DarkTeal}{Level 3: Reasoning}
\begin{itemize}
\item Why this specific decision
\item Contributing factors
\item Actionable recourse
\end{itemize}

\column{0.48\textwidth}
\textcolor{ForestGreen}{\textbf{Consent Design}}

\textcolor{Amber}{Informed Consent Requires:}
\begin{itemize}
\item Clear language (not legalese)
\item Specific purposes
\item Real choices
\item Easy opt-out
\item Granular control
\end{itemize}

\vspace{0.3cm}
\textcolor{Teal}{\textbf{GDPR Requirements:}}
\begin{itemize}
\item Right to explanation
\item Right to deletion
\item Right to object
\item Right to human review
\end{itemize}

\vspace{0.3cm}
\textcolor{DarkTeal}{\textbf{Anti-pattern:}}\\
``Agree or leave'' is not real consent
\end{columns}

\vspace{\fill}
\footnotesize\textcolor{Slate}{Transparency without agency is surveillance}
\end{frame}

% Slide 5: Case Study - Detroit Facial Recognition
\begin{frame}{Case Study: Detroit Wrongful Arrest}
\begin{columns}[T]
\column{0.48\textwidth}
\textcolor{ForestGreen}{\textbf{The Incident (2020, settled 2024)}}

\textcolor{Amber}{Facts:}
\begin{itemize}
\item Robert Williams, Black man
\item Arrested for shoplifting
\item Facial recognition false match
\item Detained 30 hours
\item Charges eventually dropped
\end{itemize}

\vspace{0.3cm}
\textcolor{Teal}{Technical Failure:}
\begin{itemize}
\item Low-quality image
\item No confidence threshold
\item No human verification
\item Known bias in FR systems
\end{itemize}

\column{0.48\textwidth}
\textcolor{ForestGreen}{\textbf{Lessons Learned}}

\textcolor{DarkTeal}{What Went Wrong:}
\begin{itemize}
\item Over-reliance on technology
\item No quality checks
\item Ignored known limitations
\item Procedural shortcuts
\item Lack of accountability
\end{itemize}

\vspace{0.3cm}
\textcolor{Amber}{Outcome:}
\begin{itemize}
\item Settlement \& policy change
\item Detroit now bans FR-only arrests
\item National attention to FR bias
\item ACLU lawsuits in multiple cities
\end{itemize}

\vspace{0.3cm}
\textcolor{Teal}{\textbf{Key Insight:}}\\
Technology amplifies existing biases in criminal justice
\end{columns}

\vspace{\fill}
\footnotesize\textcolor{Slate}{Source: ACLU, Detroit Free Press, 2024}
\end{frame}

% Slide 7: Hiring Algorithm Bias
\begin{frame}{Case Study: AI in Hiring}
\begin{columns}[T]
\column{0.48\textwidth}
\textcolor{ForestGreen}{\textbf{The Problem}}

\textcolor{Amber}{2024 Reality:}
\begin{itemize}
\item 79\% of Fortune 500 use AI screening
\item Most managers untrained
\item Women \& minorities disadvantaged
\item Black-box decision systems
\end{itemize}

\vspace{0.3cm}
\textcolor{Teal}{How Bias Enters:}
\begin{itemize}
\item Historical hiring data (biased)
\item Proxy variables (school names)
\item Resume keywords (gendered)
\item Face/voice analysis (discriminatory)
\end{itemize}

\column{0.48\textwidth}
\textcolor{ForestGreen}{\textbf{Documented Cases}}

\textcolor{DarkTeal}{Amazon (2018):}
\begin{itemize}
\item Resume screening tool
\item Penalized ``women's college''
\item Learned from male-dominated data
\item Scrapped after discovery
\end{itemize}

\vspace{0.3cm}
\textcolor{Amber}{HireVue (2021):}
\begin{itemize}
\item Video interview AI
\item Analyzed facial expressions
\item Lacked scientific validity
\item Dropped after criticism
\end{itemize}

\vspace{0.3cm}
\textcolor{Teal}{\textbf{Regulatory Response:}}\\
NYC Local Law 144 (2023)\\
Requires bias audits
\end{columns}

\vspace{\fill}
\footnotesize\textcolor{Slate}{Hiring AI perpetuates discrimination at scale}
\end{frame}

% Slide 8: Participatory Design
\begin{frame}{Participatory Design with Affected Communities}
\begin{columns}[T]
\column{0.48\textwidth}
\textcolor{ForestGreen}{\textbf{Why It Matters}}

\textcolor{Amber}{Traditional Approach:}
\begin{itemize}
\item Design for users
\item Expert-driven
\item Top-down
\item Assumptions-based
\end{itemize}

\vspace{0.3cm}
\textcolor{Teal}{Participatory Approach:}
\begin{itemize}
\item Design with users
\item Community-driven
\item Bottom-up
\item Experience-based
\end{itemize}

\vspace{0.3cm}
\textcolor{DarkTeal}{\textbf{Especially Critical:}}\\
When system affects marginalized groups without choice

\column{0.48\textwidth}
\textcolor{ForestGreen}{\textbf{Practical Methods}}

\textcolor{Amber}{Community Engagement:}
\begin{itemize}
\item Co-design workshops
\item Paid advisory boards
\item Continuous feedback
\item Shared decision-making
\end{itemize}

\textcolor{Teal}{Compensation:}
\begin{itemize}
\item Fair payment for time
\item Acknowledge expertise
\item Share ownership
\item Long-term relationships
\end{itemize}

\vspace{0.3cm}
\textcolor{DarkTeal}{\textbf{Examples:}}
\begin{itemize}
\item Detroit Digital Justice Coalition
\item Data for Black Lives
\item Indigenous data sovereignty
\end{itemize}
\end{columns}

\vspace{\fill}

% Slide 9: Value Sensitive Design
\begin{frame}{Value Sensitive Design Framework}
\begin{columns}[T]
\column{0.48\textwidth}
\textcolor{ForestGreen}{\textbf{Three Iterations}}

\textcolor{Amber}{1. Conceptual Investigation}
\begin{itemize}
\item Identify stakeholders
\item Map values \& tensions
\item Recognize trade-offs
\item Document assumptions
\end{itemize}

\textcolor{Teal}{2. Empirical Investigation}
\begin{itemize}
\item Observe actual use
\item Interview stakeholders
\item Measure impacts
\item Test hypotheses
\end{itemize}

\textcolor{DarkTeal}{3. Technical Investigation}
\begin{itemize}
\item Design for values
\item Build prototypes
\item Evaluate alternatives
\item Iterate based on feedback
\end{itemize}

\column{0.48\textwidth}
\textcolor{ForestGreen}{\textbf{Core Values to Consider}}

\textcolor{Amber}{Human Welfare:}
\begin{itemize}
\item Safety
\item Health
\item Peace
\end{itemize}

\textcolor{Teal}{Human Dignity:}
\begin{itemize}
\item Autonomy
\item Privacy
\item Non-discrimination
\end{itemize}

\textcolor{DarkTeal}{Justice:}
\begin{itemize}
\item Fairness
\item Equality
\item Access
\end{itemize}

\vspace{0.3cm}
\textcolor{Amber}{\textbf{Key Insight:}}\\
Values often conflict -- design must navigate trade-offs explicitly
\end{columns}

\vspace{\fill}
\footnotesize\textcolor{Slate}{Embed values from the start, not as afterthought}
\end{frame}

% Slide 10: Red Teaming for Ethics
\begin{frame}{Red Teaming: Adversarial Ethics Testing}
\begin{columns}[T]
\column{0.48\textwidth}
\textcolor{ForestGreen}{\textbf{What is Red Teaming?}}

\textcolor{Amber}{Origin:}
\begin{itemize}
\item Military/cybersecurity practice
\item Simulated attack scenarios
\item Find vulnerabilities before adversaries
\end{itemize}

\vspace{0.3cm}
\textcolor{Teal}{For AI Ethics:}
\begin{itemize}
\item Deliberately try to break fairness
\item Probe for hidden biases
\item Test edge cases
\item Challenge assumptions
\item Adversarial prompting
\end{itemize}

\vspace{0.3cm}
\textcolor{DarkTeal}{\textbf{Red Team Composition:}}
\begin{itemize}
\item Diverse perspectives
\item Domain experts
\item Affected community members
\item Security professionals
\end{itemize}

\column{0.48\textwidth}
\textcolor{ForestGreen}{\textbf{Red Teaming Process}}

\textcolor{Amber}{Phase 1: Threat Modeling}
\begin{itemize}
\item Identify attack surfaces
\item Map potential harms
\item Prioritize risks
\end{itemize}

\textcolor{Teal}{Phase 2: Testing}
\begin{itemize}
\item Systematic probing
\item Boundary exploration
\item Stress testing
\item Edge case generation
\end{itemize}

\textcolor{DarkTeal}{Phase 3: Reporting}
\begin{itemize}
\item Document findings
\item Assess severity
\item Recommend mitigations
\item Track remediation
\end{itemize}

\vspace{0.3cm}
\textcolor{Amber}{\textbf{Result:}}\\
More robust, ethical systems through adversarial thinking
\end{columns}

\vspace{\fill}
\footnotesize\textcolor{Slate}{Assume adversaries -- ethical and malicious}
\end{frame}
