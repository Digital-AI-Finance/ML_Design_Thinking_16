% Slide 2: Dataset Description
\begin{frame}{Workshop Dataset: 1,000 Restaurant Reviews}
\begin{columns}[T]
\column{0.48\textwidth}
\textcolor{mlblue}{\textbf{Data Format}}

\small
\texttt{review\_id: 1234}\\
\texttt{text: "Amazing food! The service was excellent..."}\\
\texttt{verified: true}

\vspace{0.3cm}
\textcolor{mlgreen}{\textbf{Characteristics:}}
\begin{itemize}
\item 100-500 words per review
\item Mix of positive/negative
\item Various writing styles
\item Different detail levels
\item Some ambiguous cases
\end{itemize}

\column{0.48\textwidth}
\textcolor{mlorange}{\textbf{Extract These Fields:}}

\small
\textbf{Required:}
\begin{itemize}
\item overall\_rating (1-5)
\item food\_quality (1-5)
\item service\_quality (1-5)
\item price\_level (cheap/moderate/expensive)
\end{itemize}

\vspace{0.3cm}
\textbf{Optional:}
\begin{itemize}
\item ambiance\_rating (1-5)
\item top\_3\_themes (array)
\item recommended\_for (array)
\end{itemize}
\end{columns}

\vspace{\fill}
\footnotesize\textcolor{mlgray}{Dataset includes 100 human-labeled examples for validation}
\end{frame}

% Slide 3: Implementation Steps
\begin{frame}{Step-by-Step Implementation Guide}
\begin{columns}[T]
\column{0.48\textwidth}
\textcolor{mlblue}{\textbf{Phase 1: Schema (15 min)}}

\small
\begin{enumerate}
\item Define JSON schema
\item Add type constraints
\item Set value ranges
\item Mark required fields
\item Test with sample data
\end{enumerate}

\vspace{0.3cm}
\textcolor{mlgreen}{\textbf{Phase 2: Prompt (15 min)}}

\small
\begin{enumerate}
\item Write extraction prompt
\item Add role definition
\item Include examples
\item Test on 5 reviews
\item Iterate to improve
\end{enumerate}

\column{0.48\textwidth}
\textcolor{mlorange}{\textbf{Phase 3: Implementation (20 min)}}

\small
\begin{enumerate}
\item Set up function calling
\item Add validation layer
\item Implement error handling
\item Test on 50 reviews
\item Fix common failures
\end{enumerate}

\vspace{0.3cm}
\textcolor{mlpurple}{\textbf{Phase 4: Validation (10 min)}}

\small
\begin{enumerate}
\item Run on 100 labeled examples
\item Calculate accuracy
\item Analyze failure cases
\item Document results
\end{enumerate}
\end{columns}

\vspace{\fill}
\footnotesize\textcolor{mlgray}{Starter notebook provided with code templates}
\end{frame}

% Slide 5: Results Analysis
\begin{frame}{Analyzing Your Results}
\begin{columns}[T]
\column{0.48\textwidth}
\textcolor{mlblue}{\textbf{What to Analyze}}

\small
\begin{enumerate}
\item Accuracy by field
\item Common error patterns
\item Confidence vs accuracy
\item Processing time
\item Cost per review
\item Edge case handling
\end{enumerate}

\vspace{0.3cm}
\textcolor{mlgreen}{\textbf{Questions to Ask:}}
\begin{itemize}
\item Which fields fail most?
\item Why did specific cases fail?
\item Is confidence score reliable?
\item What patterns emerge?
\end{itemize}

\column{0.48\textwidth}
\textcolor{mlorange}{\textbf{Iteration Strategies}}

\small
\textbf{If accuracy $<$ 90\%:}
\begin{itemize}
\item Add more examples to prompt
\item Refine schema constraints
\item Improve error handling
\item Lower temperature
\item Try chain-of-thought
\end{itemize}

\vspace{0.3cm}
\textbf{If too slow:}
\begin{itemize}
\item Remove unnecessary steps
\item Use smaller model
\item Add caching
\item Batch process
\end{itemize}
\end{columns}

\vspace{\fill}
\footnotesize\textcolor{mlgray}{Iteration is key - expect 2-3 refinement cycles}
\end{frame}
