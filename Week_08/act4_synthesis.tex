\begin{frame}[t]{Key Principles: Lessons Beyond This Specific Problem}
\small
\begin{columns}[T]
\column{0.49\textwidth}
\raggedright
\textbf{Universal Lessons}:

\vspace{0.2cm}
\textcolor{mlpurple}{\textbf{1. Structure > Power}}
\begin{itemize}
\item Smaller models with structure outperform larger models without it
\item Architecture matters more than parameters
\item Constraints enable reliability
\item Structure is feature, not limitation
\end{itemize}

\vspace{0.2cm}
\textcolor{mlblue}{\textbf{2. Validation = Reliability}}
\begin{itemize}
\item Can't improve what you can't measure
\item Validation makes failures visible
\item Visibility enables recovery
\item Recovery enables production deployment
\end{itemize}

\vspace{0.2cm}
\textcolor{mlgreen}{\textbf{3. Contracts Beat Suggestions}}
\begin{itemize}
\item Prompts are suggestions (weak)
\item Schemas are contracts (strong)
\item Enforcement at API level
\item Guaranteed structure or explicit error
\end{itemize}

\column{0.49\textwidth}
\raggedright
\textbf{Universal Lessons (continued)}:

\vspace{0.2cm}
\textcolor{mlorange}{\textbf{4. Design for Predictable Failure}}
\begin{itemize}
\item Perfect reliability is impossible
\item Predictable failure is acceptable
\item Explicit error states
\item Graceful degradation paths
\item Human-in-the-loop escalation
\end{itemize}

\vspace{0.3cm}
\textbf{Where to Apply These Principles}:
\begin{itemize}

\item Data extraction systems
\item Form automation
\item System-to-system integration
\item Production AI deployment
\item Workflow automation
\end{itemize}

\vspace{0.3cm}
\textbf{The Meta-Lesson}:
\begin{itemize}
\item These aren't specific to structured outputs
\item They apply to ANY production AI system
\item Structure, validation, contracts, predictable failure
\item Transferable to your projects
\end{itemize}
\end{columns}

\bottomnote{Four universal principles apply beyond structured outputs to any AI reliability challenge}

% Slide 18: Detail - When to Use Structured Outputs (Judgment Criteria)
\begin{frame}[t]{When to Use Structured Outputs: Judgment Criteria}
\small
\begin{columns}[T]
\column{0.49\textwidth}
\raggedright
\textcolor{mlgreen}{\textbf{When Appropriate}}:

\vspace{0.2cm}
\textbf{Production Requirements}:
\begin{itemize}
\item System integration required
\item High volume automation needed
\item Type safety is critical
\item Downstream validation essential
\item Zero-tolerance for parsing errors
\end{itemize}

\vspace{0.2cm}
\textbf{Scale Indicators}:
\begin{itemize}
\item Processing hundreds+ items daily
\item Multiple systems consuming output
\item Automated workflows dependent on data
\item No human review in every loop
\end{itemize}

\vspace{0.2cm}
\textbf{Complexity Signals}:
\begin{itemize}
\item Nested data structures
\item Multiple data types required
\item Conditional field validation
\item Cross-field dependencies
\end{itemize}

\column{0.49\textwidth}
\raggedright
\textcolor{mlred}{\textbf{When Overkill}}:

\vspace{0.2cm}
\textbf{Simple Scenarios}:
\begin{itemize}
\item One-time tasks or ad-hoc queries
\item Human review always required
\item Simple ChatGPT interactions
\item Prototyping and exploration phase
\item No downstream systems
\end{itemize}

\vspace{0.2cm}
\textbf{Low Volume}:
\begin{itemize}
\item Processing fewer than 10 items/day
\item Manual workflows acceptable
\item Flexible output formats OK
\item Quick turnaround more important
\end{itemize}

\vspace{0.2cm}
\textbf{Alternative Solutions Better}:
\begin{itemize}
\item Simple regex patterns sufficient
\item Templates work well enough
\item Cost of structure exceeds benefit
\item Requirements change frequently
\item Exploration over production
\end{itemize}
\end{columns}

\vspace{0.3cm}
\begin{center}
\textcolor{mlpurple}{\textbf{The Principle:}} Right tool for right job - structured outputs shine in production automation, not prototyping
\end{center}

\bottomnote{Judgment criteria enable wise tool selection - match solution complexity to problem requirements}
\end{frame}

\begin{center}

\vspace{0.3cm}

\begin{columns}[T]
\column{0.32\textwidth}
\small
\textcolor{mlred}{\textbf{Design Pitfalls}}
\begin{itemize}
\item Over-engineering simple tasks
\item Schema too rigid for evolution
\item Validation too strict
\item No versioning strategy
\end{itemize}

\column{0.32\textwidth}
\small
\textcolor{mlorange}{\textbf{Operational Pitfalls}}
\begin{itemize}
\item Cost blindness
\item No fallback when validation fails
\item Single point of failure
\item Inadequate error logging
\end{itemize}

\column{0.32\textwidth}
\small
\textcolor{mlblue}{\textbf{Data Pitfalls}}
\begin{itemize}
\item Ignoring edge cases
\item Insufficient testing on real data
\item No monitoring for drift
\item Missing confidence scores
\end{itemize}
\end{columns}

\vspace{0.5cm}
\begin{center}
\textcolor{mlgreen}{\textbf{Prevention:}} Test on real messy data, design for graceful degradation, monitor continuously, version schemas
\end{center}

\bottomnote{Common pitfalls are predictable and preventable - robust systems anticipate failure modes from day one}
\end{frame}
