% Slide 24: Generative model landscape (VAE, GAN, Diffusion, Transformer)
\begin{frame}
\frametitle{The Generative AI Landscape}
\framesubtitle{Four Fundamental Approaches}

\begin{center}
\includegraphics[width=0.9\textwidth]{charts/generative_landscape.pdf}
\end{center}

\vspace{0.3cm}

\begin{columns}[c]
\column{0.48\textwidth}
\textbf{Variational Autoencoders (VAEs):}
\begin{itemize}
\item Probabilistic encoding
\item Smooth latent space
\item Good for interpolation
\item Blurry outputs
\end{itemize}

\vspace{0.3cm}
\textbf{Generative Adversarial Networks (GANs):}
\begin{itemize}
\item Adversarial training
\item Sharp, realistic outputs
\item Training instability
\item Mode collapse risk
\end{itemize}

\column{0.48\textwidth}
\textbf{Diffusion Models:}
\begin{itemize}
\item Iterative denoising
\item High-quality generation
\item Stable training
\item Slow sampling
\end{itemize}

\vspace{0.3cm}
\textbf{Autoregressive Transformers:}
\begin{itemize}
\item Sequential generation
\item Excellent for text
\item Scalable architecture
\item Left-to-right bias
\end{itemize}
\end{columns}

\vspace{\fill}
\small \textcolor{gray}{
Each approach has unique strengths - modern systems often combine multiple techniques
}
\end{frame}

% Slide 25: Trade-offs: Training stability vs sample quality
\begin{frame}
\frametitle{Fundamental Trade-offs}
\framesubtitle{No Free Lunch in Generative Modeling}

\begin{center}
\includegraphics[width=0.9\textwidth]{charts/generative_tradeoffs.pdf}
\end{center}

\vspace{0.3cm}

\begin{columns}[c]
\column{0.48\textwidth}
\textbf{Training Stability:}
\begin{itemize}
\item \textcolor{mlgreen}{VAEs:} Very stable
\item \textcolor{mlgreen}{Diffusion:} Stable
\item \textcolor{mlorange}{Transformers:} Moderately stable
\item \textcolor{mlred}{GANs:} Unstable, requires tuning
\end{itemize}

\vspace{0.3cm}
\textbf{Sampling Speed:}
\begin{itemize}
\item \textcolor{mlgreen}{VAEs \& GANs:} Fast (single pass)
\item \textcolor{mlorange}{Transformers:} Moderate
\item \textcolor{mlred}{Diffusion:} Slow (1000 steps)
\end{itemize}

\column{0.48\textwidth}
\textbf{Sample Quality:}
\begin{itemize}
\item \textcolor{mlgreen}{Diffusion:} Excellent
\item \textcolor{mlgreen}{GANs:} Very good (when stable)
\item \textcolor{mlorange}{Transformers:} Good
\item \textcolor{mlred}{VAEs:} Fair (blurry)
\end{itemize}

\vspace{0.3cm}
\textbf{Controllability:}
\begin{itemize}
\item \textcolor{mlgreen}{Diffusion:} Excellent (guidance)
\item \textcolor{mlgreen}{Transformers:} Very good (prompts)
\item \textcolor{mlorange}{VAEs:} Good (interpolation)
\item \textcolor{mlred}{GANs:} Limited
\end{itemize}
\end{columns}

\vspace{\fill}
\small \textcolor{gray}{
Choose your approach based on your specific requirements: quality, speed, or stability
}
\end{frame}
