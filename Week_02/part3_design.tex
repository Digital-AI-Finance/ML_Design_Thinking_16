% Design Principles from Clustering
\begin{frame}{Design Principles Derived from Clustering}
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\Large\textbf{Universal Principles}
\normalsize
\vspace{0.5em}
\begin{enumerate}
\item \textbf{Progressive Complexity}\\
\small Start simple, reveal advanced features
\vspace{0.3em}
\item \textbf{Flexible Pathways}\\
\small Multiple routes to same goal
\vspace{0.3em}
\item \textbf{Contextual Help}\\
\small Right assistance at right time
\vspace{0.3em}
\item \textbf{Social Proof}\\
\small Show similar users' success
\vspace{0.3em}
\item \textbf{Personalized Defaults}\\
\small Smart presets per segment
\end{enumerate}
\end{column}

\begin{column}{0.48\textwidth}
\Large\textbf{Segment-Specific}
\normalsize
\vspace{0.5em}

\textcolor{mlblue}{\textbf{Beginners:}}
\begin{itemize}
\item Large buttons \& text
\item Fewer options
\item More guidance
\end{itemize}

\textcolor{mlorange}{\textbf{Advanced:}}
\begin{itemize}
\item Keyboard shortcuts
\item Batch operations
\item API access
\end{itemize}

\textcolor{mlgreen}{\textbf{Social:}}
\begin{itemize}
\item Share everywhere
\item Community features
\item Recognition systems
\end{itemize}
\end{column}
\end{columns}

\vspace{\fill}
\footnotesize\textcolor{mlgray}{Universal principles coexist with targeted application - foundational design consistency enables segment-specific adaptations without fragmentation}
\end{frame}
