\textbf{Process:}
\begin{enumerate}

\item Launch to all users
\item Cross fingers
\item Wait for complaints
\item Emergency fix if broken
\item Repeat in 12 months

\vspace{0.3cm}
\textbf{Problems:}
\begin{itemize}
\item High risk (all eggs in one basket)
\item Slow learning (feedback after 6 months)
\item No data (guessing what works)

\small
\textbf{Process:}
\begin{enumerate}
\item Form hypothesis (``X will improve Y'')
\item Design experiment (A/B test)
\item Calculate sample size (power analysis)
\item Deploy to 5\% users (low risk)
\item Measure results (data-driven)
\item Ship winners, kill losers
\item Repeat weekly
\end{enumerate}

\vspace{0.3cm}
\textbf{Benefits:}
\begin{itemize}
\item Low risk (gradual rollout)
\item Fast learning (results in days)
\item Data-driven (know what works)
\item Compound gains (many small wins)
\item Competitive edge (10$	imes$ iteration speed)
\end{itemize}

\vspace{0.3cm}
\textcolor{mlgreen}{\textbf{Success rate: 30-40\%}}

\vspace{0.2cm}

\end{columns}

\footnotesize\textcolor{mlgray}{Experimentation is not about being right—it's about learning fast}

% Slide 4: A/B Testing Fundamentals
\begin{frame}{A/B Testing Fundamentals: The Gold Standard}
\begin{columns}[T]
\column{0.48\textwidth}
\textcolor{mlblue}{\textbf{What is A/B Testing?}}

\small
\textbf{Definition:} Randomized controlled trial comparing two versions to determine which performs better.

\vspace{0.3cm}
\textbf{Core Components:}
\begin{enumerate}
\item \textbf{Control Group (A):} Current system (baseline)
\item \textbf{Treatment Group (B):} New variant (challenger)
\item \textbf{Random Assignment:} Users split 50/50
\item \textbf{Metric:} What you're measuring (CTR, revenue)
\item \textbf{Statistical Test:} Determine if difference is real
\end{enumerate}

\vspace{0.3cm}
\textcolor{mlgreen}{\textbf{Example: Recommendation Algorithm}}

\small
\begin{itemize}
\item Control: Popularity-based (5\% CTR)
\item Treatment: ML collaborative filtering
\item Metric: Click-through rate
\item Users: 50,000 per group
\item Duration: 2 weeks
\item Result: Treatment wins (6.2\% CTR)
\end{itemize}

\column{0.48\textwidth}
\textcolor{mlorange}{\textbf{Why Randomization Matters}}

\small
\textbf{Without randomization:}
\begin{itemize}
\item Selection bias (power users in treatment)
\item Confounding variables (time of day, device)
\item Can't isolate causal effect
\item Results are unreliable
\end{itemize}

\vspace{0.3cm}
\textbf{With randomization:}
\begin{itemize}
\item Groups are statistically equivalent
\item Only difference is the treatment
\item Causal interpretation valid
\item Results are trustworthy
\end{itemize}

\vspace{0.3cm}
\textcolor{mlpurple}{\textbf{Key Requirements}}

\small
\begin{itemize}
\item \textbf{Sample size:} Large enough for statistical power
\item \textbf{Duration:} Long enough to capture true behavior
\item \textbf{Randomization:} Truly random assignment
\item \textbf{Isolation:} No spillover between groups
\item \textbf{Pre-registration:} Hypothesis before experiment
\end{itemize}
\end{columns}

\vspace{\fill}
\footnotesize\textcolor{mlgray}{A/B testing: The scientific method applied to product development}
\end{frame}

% Slide 6: Real Failure Examples
\begin{frame}{Real Failure Examples: The Cost of Not Testing}
\begin{columns}[T]
\column{0.48\textwidth}
\textcolor{mlred}{\textbf{Knight Capital: \$440M Loss}}

\small
\textbf{What Happened:}
\begin{itemize}
\item August 1, 2012: Deploy new trading algorithm
\item No A/B test, no gradual rollout
\item Bug: Bought high, sold low repeatedly
\item Duration: 45 minutes
\item Loss: \$440 million
\item Outcome: Company bankrupt
\end{itemize}

\vspace{0.3cm}
\textcolor{mlblue}{\textbf{What Should Have Been Done:}}
\begin{itemize}
\item Shadow mode testing (parallel run)
\item Canary deployment (1\% traffic first)
\item Kill switch (automatic rollback)
\item Guardrail metrics (loss limits)
\end{itemize}

\vspace{0.3cm}
\textbf{Lesson:} High-stakes deployments demand rigorous testing

\column{0.48\textwidth}
\textcolor{mlorange}{\textbf{Microsoft Tay Chatbot}}

\small
\textbf{What Happened:}
\begin{itemize}
\item March 2016: Launch AI chatbot on Twitter
\item No gradual rollout, no guardrails
\item Users exploited vulnerabilities
\item Within 16 hours: Racist, offensive tweets
\item Outcome: Shut down, PR disaster
\end{itemize}

\vspace{0.3cm}
\textcolor{mlpurple}{\textbf{What Should Have Been Done:}}
\begin{itemize}
\item Beta test with controlled users
\item Content moderation safeguards
\item Incremental rollout (100 $\rightarrow$ 1K $\rightarrow$ 10K users)
\item Real-time monitoring and kill switch
\end{itemize}

\vspace{0.3cm}
\textbf{Lesson:} AI in the wild requires safety testing

\vspace{0.3cm}
\textcolor{mlgreen}{\textbf{Common Thread}}

\small
Both failures share: (1) No gradual rollout, (2) No guardrails, (3) No monitoring, (4) No rollback plan
\end{columns}

\vspace{\fill}
\footnotesize\textcolor{mlgray}{A/B testing isn't just about winning—it's about avoiding catastrophic losses}
\end{frame}

% Slide 8: When to A/B Test ML Models
\begin{frame}{When to A/B Test ML Models}
\begin{columns}[T]
\column{0.48\textwidth}
\textcolor{mlblue}{\textbf{Always A/B Test}}

\small
\textbf{1. New Model Deployment}
\begin{itemize}
\item Replacing existing model
\item High risk of regression
\item Need causal validation
\end{itemize}

\vspace{0.2cm}
\textbf{2. Algorithm Changes}
\begin{itemize}
\item Random Forest $

\rightarrow$ XGBoost
\item Content-based $

\rightarrow$ collaborative filtering
\item Offline metrics don't predict online
\end{itemize}

\vspace{0.2cm}
\textbf{3. Feature Updates}
\begin{itemize}
\item Add new features
\item Remove stale features
\item Feature engineering changes
\end{itemize}

\vspace{0.2cm}
\textbf{4. Hyperparameter Tuning}
\begin{itemize}
\item Learning rate changes
\item Regularization adjustments
\item May help or hurt in production
\end{itemize}

\vspace{0.2cm}
\textbf{5. Retraining Cadence}
\begin{itemize}
\item Daily vs weekly vs monthly
\item Balance freshness and stability
\end{itemize}

\column{0.48\textwidth}
\textcolor{mlorange}{\textbf{Sometimes A/B Test}}

\small
\textbf{6. UI/UX Changes}
\begin{itemize}
\item Model recommendations display
\item Explanation text
\item Confidence thresholds shown to users
\end{itemize}

\vspace{0.2cm}
\textbf{7. Business Logic}
\begin{itemize}
\item Ranking vs filtering
\item Diversity vs relevance
\item Personalization intensity
\end{itemize}

\vspace{0.3cm}
\textcolor{mlgreen}{\textbf{Don't Bother A/B Testing}}

\small
\textbf{8. Bug Fixes}
\begin{itemize}
\item Obvious correctness issues
\item Fix immediately, no test needed
\end{itemize}

\vspace{0.2cm}
\textbf{9. Infrastructure Changes}
\begin{itemize}
\item Same model, faster serving
\item No user-facing behavior change
\end{itemize}

\vspace{0.2cm}
\textcolor{mlpurple}{\textbf{Rule of Thumb}}

\small
If change affects user experience or business metrics $\rightarrow$ A/B test it
\end{columns}

\vspace{\fill}
\footnotesize\textcolor{mlgray}{When in doubt, test it. Cost of experiment $<$ cost of bad deployment}
\end{frame}
