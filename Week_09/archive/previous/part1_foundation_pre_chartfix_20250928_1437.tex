% Slide 4: When Accuracy Fails
\begin{frame}{Four Scenarios Where Accuracy Fails}
\begin{columns}[T]
\column{0.48\textwidth}
\textcolor{mlred}{\textbf{1. Class Imbalance}}

\small
\textbf{Problem:} Rare event detection\\
\textbf{Example:} Fraud (0.1\% of transactions)\\
\textbf{Why:} Predict majority class = high accuracy\\
\textbf{Fix:} Use Precision-Recall curves

\vspace{0.3cm}
\textcolor{mlorange}{\textbf{2. Cost Asymmetry}}

\small
\textbf{Problem:} Error costs differ\\
\textbf{Example:} Medical false negative $>$ false positive\\
\textbf{Why:} Accuracy treats errors equally\\
\textbf{Fix:} Weighted metrics, business ROI

\column{0.48\textwidth}
\textcolor{mlgreen}{\textbf{3. Threshold Sensitivity}}

\small
\textbf{Problem:} Performance varies by cutoff\\
\textbf{Example:} 0.5 threshold arbitrary\\
\textbf{Why:} Accuracy hides threshold impact\\
\textbf{Fix:} ROC curves, threshold optimization

\vspace{0.3cm}
\textcolor{mlpurple}{\textbf{4. Multi-Class Confusion}}

\small
\textbf{Problem:} Some class pairs matter more\\
\textbf{Example:} Confusing cat/dog ok, cat/car bad\\
\textbf{Why:} Accuracy averages all errors\\
\textbf{Fix:} Confusion matrix analysis
\end{columns}

\vspace{\fill}
\footnotesize\textcolor{mlgray}{Each scenario requires specific metrics beyond accuracy}
\end{frame}

% Slide 5: Confusion Matrix Deep Dive
\begin{frame}{Confusion Matrix: Four Numbers, Infinite Insights}
\begin{columns}[T]
\column{0.55\textwidth}
\includegraphics[width=\textwidth]{charts/confusion_matrix_anatomy.pdf}

\column{0.43\textwidth}
\textcolor{mlblue}{\textbf{The Four Quadrants}}

\small
\textbf{True Positive (TP):}\\
Predicted positive, actually positive\\
Example: Detected fraud that was real

\vspace{0.2cm}
\textbf{True Negative (TN):}\\
Predicted negative, actually negative\\
Example: Approved legitimate transaction

\vspace{0.2cm}
\textbf{False Positive (FP):}\\
Predicted positive, actually negative\\
Example: Blocked legitimate transaction

\vspace{0.2cm}
\textbf{False Negative (FN):}\\
Predicted negative, actually positive\\
Example: Missed actual fraud
\end{columns}

\vspace{\fill}
\footnotesize\textcolor{mlgray}{All metrics derive from these four numbers}
\end{frame}

% Slide 6: The Precision-Recall Spectrum
\begin{frame}{Precision vs Recall: The Fundamental Trade-Off}
\begin{columns}[T]
\column{0.48\textwidth}
\textcolor{mlblue}{\textbf{Precision}}

\small
\textbf{Definition:} TP / (TP + FP)\\
\textbf{Question:} When I predict positive, am I right?\\
\textbf{Focus:} Accuracy of positive predictions

\vspace{0.3cm}
\textbf{High Precision Means:}
\begin{itemize}
\item Few false alarms
\item Conservative predictions
\item High confidence when predicting positive
\end{itemize}

\vspace{0.3cm}
\textcolor{mlgray}{\small Use when:} False positives costly\\
\textcolor{mlgray}{\small Example:} Email spam (don't block real email)

\column{0.48\textwidth}
\textcolor{mlorange}{\textbf{Recall (Sensitivity)}}

\small
\textbf{Definition:} TP / (TP + FN)\\
\textbf{Question:} Of all actual positives, how many did I catch?\\
\textbf{Focus:} Completeness of detection

\vspace{0.3cm}
\textbf{High Recall Means:}
\begin{itemize}
\item Catch most positive cases
\item Aggressive predictions
\item Few missed detections
\end{itemize}

\vspace{0.3cm}
\textcolor{mlgray}{\small Use when:} False negatives costly\\
\textcolor{mlgray}{\small Example:} Disease screening (don't miss sick patients)
\end{columns}

\vspace{\fill}
\footnotesize\textcolor{mlgray}{You cannot optimize both simultaneously - trade-off is fundamental}
\end{frame}
